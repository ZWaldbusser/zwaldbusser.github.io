/**
 * 
 * This component displays a dashboard for managers with various interactive boxes representing
 * different management functionalities like Employee Manager, Data Dashboard, Inventory Manager, etc.
 * 
 * @author Zeina Abu Ghosh
 */
import './managerPage.css';
import React, { useState, useEffect } from 'react';
import { executeQuery } from './db';

/**
 * Creates a simple styled box component for displaying static information.
 * @param {string} name - The label text displayed inside the box.
 * @param {string} size - The size of the box (width and height) in CSS units.
 * @returns {JSX.Element} A styled box with a label.
 */
function MyBox({name, size}) {
  return (
    <div style={{ width: size, height: size, margin: '20px', backgroundColor: 'lightgrey' }}>
      <center>
        <h2 style={{ fontFamily: 'Arial, sans-serif' }}>{name}</h2>
        
      </center>
    </div>
  );
}
/**
 * Creates a dynamic box that fetches and displays data from the database.
 * @param {string} name - The ID used to query the database for the item's name.
 * @param {string} size - The size of the box (width and height) in CSS units.
 * @returns {JSX.Element} A styled box with the fetched data or an error message.
 */
function TestBox({ name, size}) {
 
  const [actualName, setActualName] = useState('');

  /**
   * Fetches the menu item name from the database.
   * @async
   * @function
   * @throws {Error} Displays 'error' if the database query fails.
   */
  //Working sql query. dbServer must be run in order to establish a connection
  useEffect(() => {
    async function fetchMenuItem() {

      const query = 'SELECT name FROM menu_items WHERE menu_item_id = $1;';
      const params = [name];  // Use 'name' as menu_item_id
      try{
        const result = await executeQuery(query, params);
        setActualName(result[0].name); //result[row#].COLUMN
      } catch (error) {

        setActualName('error');
      }
  }

    fetchMenuItem();
  }, [name]);

  return (
    <div style={{ width: size, height: size, margin: '20px', backgroundColor: 'lightgrey' }}>
      <center>
        <h2 style={{ fontFamily: 'Arial, sans-serif' }}>{actualName}</h2>
      </center>
    </div>
  );
  }

/**
 * Creates the Manager Page, which displays different management functionalities
 * such as Employee Manager, Data Dashboard, Inventory Manager, and dynamic data boxes.
 * @returns {JSX.Element} Returns the JSX for the Manager Page layout.
 */
export default function ManagerPage() {
  return (
    <div className="full-background" style={{ display: 'flex', justifyContent: 'center', alignItems: 'center' }}>
      <MyBox name="Employee Manager" size='400px'/>
      <MyBox name="Data Dashboard" size='400px'/>
      <MyBox name="Inventory Manager" size='400px'/>
      <TestBox name="1" size='200px'/>
      <TestBox name="2" size='200px'/>
    </div>
    
  );
}
